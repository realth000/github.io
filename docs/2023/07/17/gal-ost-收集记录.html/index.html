#!/bin/bash

for f in $(ls *.ogg)
do
  match=$(echo $f | sed -n "s/bgm\(.*\)_a\.ogg/\1/p")
  if [ "x$match" != "x" ];then
    echo "make: \"$match\""
  fi
  name1="bgm${match}_a.ogg"
  name2="bgm${match}_b.ogg"

  if [ ! -f $name1 ] || [ ! -f $name2 ];then
    echo "skip $match"
    continue
  fi

  name3="bgm${match}.ogg"
  echo "file '${name1}'" > input.txt
  echo "file '${name2}'" >> input.txt
  echo y | ffmpeg -f concat -i input.txt -c copy $name3

done
